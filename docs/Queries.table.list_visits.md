# List of visits

```sql
CREATE TABLE `list_visits` (
 `id` INT(10) NOT NULL AUTO_INCREMENT,
 `ip` VARCHAR(50) NOT NULL DEFAULT '1' COLLATE 'utf8mb3_general_ci',
 `view` BIGINT(19) NOT NULL DEFAULT '1',
 `locale` VARCHAR(5) NOT NULL DEFAULT 'en' COLLATE 'utf8mb3_general_ci',
 `lang` VARCHAR(5) NOT NULL DEFAULT 'en' COLLATE 'utf8mb3_general_ci',
 `modified` DATETIME NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
 `created` DATETIME NULL DEFAULT CURRENT_TIMESTAMP,
 PRIMARY KEY (`id`) USING BTREE,
 UNIQUE INDEX `unique` (`ip`) USING BTREE
)
